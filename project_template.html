Visualization

import matplotlib.pyplot as plt

country_code = 'USA'
gdp_pc_country = gdp_pc.loc[country_code].dropna()

plt.figure()
plt.plot(gdp_pc_country.index, gdp_pc_country.values)
plt.xlabel("Year")
plt.ylabel("GDP per capita (constant 2015 US$)")
plt.title(f"GDP per capita over time: {country_code}")
plt.tight_layout()
